import os
import asyncio
import tempfile
import shutil
from aiogram import Bot, Dispatcher, types, F
from aiogram.filters import CommandStart
from aiogram.types import InlineKeyboardMarkup, InlineKeyboardButton
import instaloader

BOT_TOKEN = os.getenv("BOT_TOKEN")
CHANNEL_ID = int(os.getenv("CHANNEL_ID", "-1003988858235"))
CHANNEL_LINK = os.getenv("CHANNEL_LINK", "https://t.me/+yOtmhRjJeZJmMTRi")

bot = Bot(token=BOT_TOKEN)
dp = Dispatcher()

async def check_subscription(user_id: int) -> bool:
    try:
        member = await bot.get_chat_member(CHANNEL_ID, user_id)
        return member.status in ["member", "administrator", "creator"]
    except:
        return False

def get_subscribe_keyboard():
    return InlineKeyboardMarkup(inline_keyboard=[
        [InlineKeyboardButton(text="📢 Kanalga a'zo bo'lish", url=CHANNEL_LINK)],
        [InlineKeyboardButton(text="✅ A'zo bo'ldim", callback_data="check_sub")]
    ])

@dp.message(CommandStart())
async def start(message: types.Message):
    if not await check_subscription(message.from_user.id):
        await message.answer(
            "👋 Salom! Botdan foydalanish uchun kanalimizga a'zo bo'ling:",
            reply_markup=get_subscribe_keyboard()
        )
        return
    await message.answer(
        "✅ Salom! Instagram video yoki rasm havolasini yuboring — yuklab beraman!"
    )

@dp.callback_query(F.data == "check_sub")
async def check_sub_callback(callback: types.CallbackQuery):
    if await check_subscription(callback.from_user.id):
        await callback.message.edit_text(
            "✅ Rahmat! Endi Instagram havolasini yuboring."
        )
    else:
        await callback.answer("❌ Siz hali a'zo emassiz!", show_alert=True)

@dp.message(F.text)
async def handle_link(message: types.Message):
    if not await check_subscription(message.from_user.id):
        await message.answer(
            "❌ Avval kanalga a'zo bo'ling:",
            reply_markup=get_subscribe_keyboard()
        )
        return

    url = message.text.strip()
    if "instagram.com" not in url:
        await message.answer("❗ Iltimos, Instagram havolasini yuboring.")
        return

    wait_msg = await message.answer("⏳ Yuklanmoqda...")

    tmpdir = tempfile.mkdtemp()
    try:
        L = instaloader.Instaloader(
            download_videos=True,
            download_video_thumbnails=False,
            download_geotags=False,
            download_comments=False,
            save_metadata=False,
            post_metadata_txt_pattern="",
            quiet=True,
            dirname_pattern=tmpdir
        )

        shortcode = None
        if "/p/" in url:
            shortcode = url.split("/p/")[1].split("/")[0].split("?")[0]
        elif "/reel/" in url:
            shortcode = url.split("/reel/")[1].split("/")[0].split("?")[0]
        elif "/tv/" in url:
            shortcode = url.split("/tv/")[1].split("/")[0].split("?")[0]

        if not shortcode:
            await wait_msg.edit_text("❗ Havola noto'g'ri. Instagram post havolasini yuboring.")
            return

        post = instaloader.Post.from_shortcode(L.context, shortcode)
        L.download_post(post, target=tmpdir)

        sent = False
        for root, dirs, files in os.walk(tmpdir):
            for file in files:
                filepath = os.path.join(root, file)
                if file.endswith(".mp4"):
                    with open(filepath, "rb") as f:
                        await message.answer_video(f, caption="📥 @insta_saveee_bot")
                    sent = True
                elif file.endswith((".jpg", ".jpeg", ".png")):
                    with open(filepath, "rb") as f:
                        await message.answer_photo(f, caption="📥 @insta_saveee_bot")
                    sent = True

        if not sent:
            await wait_msg.edit_text("❗ Fayl topilmadi. Havola ochiq postga tegishli bo'lishi kerak.")
        else:
            await wait_msg.delete()

    except Exception as e:
        await wait_msg.edit_text(f"❌ Xatolik: Post yopiq yoki havola noto'g'ri bo'lishi mumkin.")
    finally:
        shutil.rmtree(tmpdir, ignore_errors=True)

async def main():
    await dp.start_polling(bot)

if __name__ == "__main__":
    asyncio.run(main())
